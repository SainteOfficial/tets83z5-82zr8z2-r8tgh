local allEnemiesToggle = AutoFarmTab:CreateToggle({
    Name = "üîò All Enemies",

    CurrentValue = true,
    Callback = function(Value)
        if Value then
            -- If this toggle is activated, deactivate the boss-only toggle

            local lighting = game:GetService("Lighting")
