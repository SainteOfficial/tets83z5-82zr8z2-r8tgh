-- Luna Interface Suite laden
local success, Luna = pcall(function()
    return loadstring(game:HttpGet("https://raw.githubusercontent.com/Nebula-Softworks/Luna-Interface-Suite/refs/heads/main/source.lua", true))()
end)

if not success then
    warn("Luna Interface Suite konnte nicht geladen werden!")
    return
end

-- Fenster erstellen
local Window = Luna:CreateWindow({
    Name = "Einfaches Script Hub", 
    Subtitle = "v1.0",
    LogoID = nil, 
    LoadingEnabled = true,
    LoadingTitle = "Script Hub wird geladen...",
    LoadingSubtitle = "Scripte werden vorbereitet",
    
    ConfigSettings = {
        ConfigFolder = "MeinScriptHub"
    },
    
    KeySystem = false
})

-- Universal Tab erstellen (mit pcall für Fehlerbehandlung)
local UniversalTab = Window:CreateTab({
    Name = "Universal",
    Icon = "globe",
    ImageSource = "Material",
    ShowTitle = true
})

-- Tab-Inhaltstest
UniversalTab:CreateParagraph({
    Title = "Tab Test", 
    Text = "Wenn du diesen Text siehst, funktionieren die Tabs korrekt!"
})

-- Geschwindigkeit & Sprung Sektion
UniversalTab:CreateSection("Geschwindigkeit & Sprung")

-- WalkSpeed Toggle
UniversalTab:CreateToggle({
    Name = "Schneller Laufen",
    Description = "Erhöht deine Laufgeschwindigkeit",
    CurrentValue = false,
    Callback = function(Value)
        pcall(function()
            if Value then
                game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = 50
            else
                game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = 16
            end
        end)
    end
}, "WalkSpeedToggle")

-- Einstellungen Tab
local SettingsTab = Window:CreateTab({
    Name = "Einstellungen",
    Icon = "settings",
    ImageSource = "Material",
    ShowTitle = true
})

-- Theming Sektion hinzufügen
pcall(function() SettingsTab:BuildThemeSection() end)

-- Konfigurationssektion hinzufügen
pcall(function() SettingsTab:BuildConfigSection() end)

-- WICHTIG: Automatische Konfiguration laden (muss am Ende sein!)
Luna:LoadAutoloadConfig()
